      let navCurrentX = 0;

      function lerpNav() {
        navCurrentX += (navTargetX - navCurrentX) * 0.18;
        const cx = navCurrentX;
        const container = document.getElementById("bottom-nav-container");
        const w = container
          ? container.clientWidth
          : window.innerWidth > 480
            ? 480
            : window.innerWidth;

        const fR = 14;
        const cR = 32;
        const fX = 45.11;
        const tX = 31.38;
        const tY = 11.26;

        const path = `M 0 0 L ${cx - fX} 0 A ${fR} ${fR} 0 0 1 ${cx - tX} ${tY} A ${cR} ${cR} 0 0 0 ${cx + tX} ${tY} A ${fR} ${fR} 0 0 1 ${cx + fX} 0 L ${w} 0 L ${w} 128 L 0 128 Z`;
        document.getElementById("nav-path").setAttribute("d", path);

        document.getElementById("nav-dot").style.transform =
          `translateX(${cx - 25}px)`;

        if (Math.abs(navTargetX - navCurrentX) > 0.1) {
          navRaf = requestAnimationFrame(lerpNav);
        } else {
          navCurrentX = navTargetX;
          const finalPath = `M 0 0 L ${navTargetX - fX} 0 A ${fR} ${fR} 0 0 1 ${navTargetX - tX} ${tY} A ${cR} ${cR} 0 0 0 ${navTargetX + tX} ${tY} A ${fR} ${fR} 0 0 1 ${navTargetX + fX} 0 L ${w} 0 L ${w} 128 L 0 128 Z`;
          document.getElementById("nav-path").setAttribute("d", finalPath);
          document.getElementById("nav-dot").style.transform =
            `translateX(${navTargetX - 25}px)`;
          navRaf = null;
        }
      }

      function switchNavSet(targetId) {
        const allIds = ["main-nav-items", "about-nav-items", "pricing-nav-items", "project-nav-items"];
        
        allIds.forEach(id => {
          const el = document.getElementById(id);
          if (!el) return;
          
          if (id === targetId) {
            if (el.classList.contains("nav-set-active") && !el.classList.contains("hidden")) {
              return;
            }
            // Prepare for sliding in from right
            el.classList.remove("hidden");
            el.classList.remove("nav-set-active", "nav-set-hide-down");
            el.classList.add("nav-set-from-right");
            
            // Force layout reflow
            void el.offsetWidth;
            
            // Start the transition
            el.classList.remove("nav-set-from-right");
            el.classList.add("nav-set-active");
          } else {
            if (el.classList.contains("nav-set-active") && !el.classList.contains("hidden")) {
              // Sliding down & fading out
              el.classList.remove("nav-set-active");
              el.classList.add("nav-set-hide-down");
              
              // Hide after transition duration
              setTimeout(() => {
                if (!el.classList.contains("nav-set-active")) {
                  el.classList.add("hidden");
                }
              }, 450);
            } else {
              el.classList.add("hidden");
              el.classList.remove("nav-set-active", "nav-set-hide-down", "nav-set-from-right");
            }
          }
        });
      }

      function setNavActive(idx, isInstant = false) {
        // If it's a modal (2 = Search, 3 = More) and we click it, don't necessarily stay there after closing?
        // Let's just track it directly
        const activeNavId = getActiveNavId();
        const items = document.querySelectorAll(`#${activeNavId} > .nav-item`);
        if (!items || items.length === 0 || !items[idx]) return;
        items.forEach((item, i) => {
          if (i === idx) item.classList.add("active");
          else item.classList.remove("active");
        });

        const navIconsEl = document.getElementById("nav-icons");
        if (!navIconsEl) return;

        const activeItem = items[idx];
        const container = document.getElementById("bottom-nav-container");
        let offsetLeft = activeItem.offsetLeft;
        let parent = activeItem.offsetParent;
        while (parent && parent !== container) {
          offsetLeft += parent.offsetLeft;
          parent = parent.offsetParent;
        }
        navTargetX = offsetLeft + activeItem.offsetWidth / 2;

        if (isInstant) {
          navCurrentX = navTargetX;
        }
        if (!navRaf) lerpNav();

        // Routing
        if (!isInstant) {
          // Don't trigger on initial load
          if (activeNavId === "main-nav-items") {
            if (idx === 0) {
              lastMainRoute = 0;
              goToLanding();
              if (searchOpen) toggleSearch(false);
              if (moreOpen) toggleMore(false);
            } else if (idx === 1) {
              lastMainRoute = 1;
              goToDashboard();
              if (searchOpen) toggleSearch(false);
              if (moreOpen) toggleMore(false);
            } else if (idx === 2) {
              // Talk route
              lastMainRoute = 2;
              goToTalk();
              if (searchOpen) toggleSearch(false);
              if (moreOpen) toggleMore(false);
            } else if (idx === 3) {
              if (searchOpen) {
                setNavActive(1);
                return;
              }
              if (!searchOpen) toggleSearch(false);
              if (moreOpen) toggleMore(false);
            } else if (idx === 4) {
              if (moreOpen) {
                setNavActive(1);
                return;
              }
              if (!moreOpen) toggleMore(false);
              if (searchOpen) toggleSearch(false);
            }
          }
        }
      }

      const sliderBtn = document.getElementById("slider-btn");

      let isDragging = false;
      let startY = 0;
      let btnBottom = 209; // corresponds to bottom-[209px]

      function handleDragStart(y) {
        if (!sliderBtn) return;
        isDragging = true;
        startY = y;
        sliderBtn.style.transition = "none";
        const arrows = document.getElementById("slider-arrows");
        if (arrows) arrows.style.transition = "none";
      }

      function handleDragMove(y) {
        if (!isDragging || !sliderBtn || !sliderTrack) return;
        const deltaY = startY - y;
        let newBottom = btnBottom + deltaY;

        const trackHeight = sliderTrack.clientHeight;
        const btnHeight = sliderBtn.clientHeight;
        const maxBottom = trackHeight - btnHeight - 20;

        if (newBottom < btnBottom) newBottom = btnBottom;

        let shouldGo = false;
        if (newBottom > maxBottom) {
          newBottom = maxBottom;
          isDragging = false; // Successfully reached top
          shouldGo = true;
        }

        sliderBtn.style.bottom = `${newBottom}px`;

        const progress = Math.max(
          0,
          Math.min(1, (newBottom - btnBottom) / (maxBottom - btnBottom)),
        );
        sliderBtn.style.opacity = 1 - progress;
        const arrows = document.getElementById("slider-arrows");
        if (arrows) arrows.style.opacity = (1 - progress) * 0.8;

        if (shouldGo) goToDashboard();
      }

      function handleDragEnd() {
        if (!isDragging || !sliderBtn) return;
        isDragging = false;
        sliderBtn.style.transition =
          "bottom 0.3s cubic-bezier(0.34, 1.56, 0.64, 1), opacity 0.3s";
        sliderBtn.style.bottom = `${btnBottom}px`;
        sliderBtn.style.opacity = 1;
        const arrows = document.getElementById("slider-arrows");
        if (arrows) {
          arrows.style.transition = "opacity 0.3s";
          arrows.style.opacity = 0.8;
          setTimeout(() => {
            if (!isDragging) arrows.style.transition = "none";
          }, 300);
        }
      }

      // Mouse events
      const landingPageNode = document.getElementById("landing");
      if (landingPageNode) {
        landingPageNode.addEventListener("mousedown", (e) => {
          handleDragStart(e.clientY);
        });
      }
      window.addEventListener("mousemove", (e) => {
        handleDragMove(e.clientY);
      });
      window.addEventListener("mouseup", handleDragEnd);

      // Touch events
      if (landingPageNode) {
        landingPageNode.addEventListener(
          "touchstart",
          (e) => {
            handleDragStart(e.touches[0].clientY);
          },
          { passive: false },
        );
      }
      window.addEventListener(
        "touchmove",
        (e) => {
          if (isDragging) {
            e.preventDefault();
            handleDragMove(e.touches[0].clientY);
          }
        },
        { passive: false },
      );
      window.addEventListener("touchend", handleDragEnd);

      function syncNavUI(idx) {
        const activeNavId = getActiveNavId();
        const items = document.querySelectorAll(
          `#${activeNavId} > .nav-item`,
        );
        if (!items || items.length === 0 || !items[idx]) return;
        items.forEach((item, i) => {
          if (i === idx) item.classList.add("active");
          else item.classList.remove("active");
        });
        const activeItem = items[idx];
        const container = document.getElementById("bottom-nav-container");
        let offsetLeft = activeItem.offsetLeft;
        let parent = activeItem.offsetParent;
        while (parent && parent !== container) {
          offsetLeft += parent.offsetLeft;
          parent = parent.offsetParent;
        }
        navTargetX = offsetLeft + activeItem.offsetWidth / 2;
        if (!navRaf) lerpNav();
      }

      window.addEventListener("resize", () => {
        const activeNavId = getActiveNavId();
        const items = document.querySelectorAll(
          `#${activeNavId} > .nav-item`,
        );
        let activeIdx = lastMainRoute;
        items.forEach((item, i) => {
          if (item.classList.contains("active")) activeIdx = i;
        });
        syncNavUI(activeIdx);
        updateProjectDetailNavShape();
      });

      function goToDashboard() {
        lastMainRoute = 1;
        const landing = document.getElementById("landing");
        if (landing) {
            landing.style.transform = "translateY(-100%)";
            landing.style.opacity = "0";
        }
        syncNavUI(1);
        const nav = document.getElementById("bottom-nav-container");
        if (nav) {
          nav.style.transform = "translateY(0)";
          nav.style.opacity = "1";
        }
      }

      function goToTalk() {
        lastMainRoute = 2;
        const landing = document.getElementById("landing");
        if (landing) {
            landing.style.transform = "translateY(-100%)";
            landing.style.opacity = "0";
        }
        syncNavUI(2);
        const nav = document.getElementById("bottom-nav-container");
        if (nav) {
          nav.style.transform = "translateY(0)";
          nav.style.opacity = "1";
        }
      }

      // Swipe down on dashboard to return
      const dashboardScroll = document.getElementById("dashboard-scroll");
      let dashStartY = 0;
      let dashIsDragging = false;

      function handleDashDragStart(y) {
        if (dashboardScroll && dashboardScroll.scrollTop <= 0) {
          dashStartY = y;
          dashIsDragging = true;
        }
      }

      function handleDashDragMove(y) {
        if (!dashIsDragging) return;
        const deltaY = y - dashStartY;
        if (deltaY > 80) {
          goToLanding();
          dashIsDragging = false;
        }
      }

      function handleDashDragEnd() {
        dashIsDragging = false;
      }

      if (dashboardScroll) {
          dashboardScroll.addEventListener("mousedown", (e) =>
            handleDashDragStart(e.clientY),
          );
      }
      window.addEventListener("mousemove", (e) => {
        if (dashIsDragging) handleDashDragMove(e.clientY);
      });
      window.addEventListener("mouseup", handleDashDragEnd);

      if (dashboardScroll) {
          dashboardScroll.addEventListener(
            "touchstart",
            (e) => handleDashDragStart(e.touches[0].clientY),
            { passive: true },
          );
          dashboardScroll.addEventListener(
            "touchmove",
            (e) => {
              if (dashIsDragging) handleDashDragMove(e.touches[0].clientY);
            },
            { passive: true },
          );
          dashboardScroll.addEventListener("touchend", handleDashDragEnd);
      }

      function goToLanding() {
        lastMainRoute = 0;
        const landing = document.getElementById("landing");
        if (landing) {
            landing.style.transform = "translateY(0)";
            landing.style.opacity = "1";
        }
        syncNavUI(0);

        const nav = document.getElementById("bottom-nav-container");
        if (nav) {
          nav.style.transform = "translateY(100px)";
          nav.style.opacity = "0";
        }

        setTimeout(() => {
          if (sliderBtn) {
              sliderBtn.style.transition = "none";
              sliderBtn.style.bottom = `${btnBottom}px`;
              sliderBtn.style.opacity = 1;
          }
          const arrows = document.getElementById("slider-arrows");
          if (arrows) {
            arrows.style.transition = "none";
            arrows.style.opacity = 0.8;
          }
        }, 600);
      }

      // Drag Scroll Logic
      document.querySelectorAll(".drag-scroll").forEach((slider) => {
        let isDown = false;
        let startX;
        let scrollLeft;
        let resumeTimeout = null;

        const startDrag = (x) => {
          isDown = true;
          slider.classList.add("active-drag");
          slider.dataset.pauseAnim = "true";
          if (resumeTimeout) clearTimeout(resumeTimeout);
          startX = x - slider.offsetLeft;
          scrollLeft = slider.scrollLeft;
        };

        const endDrag = () => {
          if (!isDown) return;
          isDown = false;
          slider.classList.remove("active-drag");
          resumeTimeout = setTimeout(() => {
            slider.dataset.pauseAnim = "false";
          }, 2000);
        };

        const moveDrag = (x, e) => {
          if (!isDown) return;
          e.preventDefault();
          const walk = (x - slider.offsetLeft - startX) * 2;
          slider.scrollLeft = scrollLeft - walk;
        };

        slider.addEventListener("mousedown", (e) => startDrag(e.pageX));
        slider.addEventListener("mouseleave", endDrag);
        slider.addEventListener("mouseup", endDrag);
        slider.addEventListener("mousemove", (e) => moveDrag(e.pageX, e));

        slider.addEventListener(
          "touchstart",
          (e) => startDrag(e.touches[0].pageX),
          { passive: true },
        );
        slider.addEventListener("touchend", endDrag);
        slider.addEventListener("touchcancel", endDrag);
        slider.addEventListener(
          "touchmove",
          (e) => {
            if (isDown && e.cancelable) e.preventDefault();
            moveDrag(e.touches[0].pageX, e);
          },
          { passive: false },
        );
      });

      // Auto-scrollers
      const autoScrollers = [
        { id: "main-carousel", setWidth: (280 + 16) * 3, dir: 1, speed: 0.3 },
        {
          id: "featured-carousel",
          setWidth: (140 + 16) * 10,
          dir: -1,
          speed: 0.3,
        },
        { id: "trusted-carousel", setWidth: 240, dir: -1, speed: 0.2 },
      ];

      autoScrollers.forEach((conf) => {
        const el = document.getElementById(conf.id);
        if (!el) return;

        let exactScrollLeft = conf.setWidth;
        el.scrollLeft = exactScrollLeft;

        el.addEventListener("scroll", () => {
          if (el.scrollLeft >= conf.setWidth * 2) {
            el.scrollLeft -= conf.setWidth;
            exactScrollLeft -= conf.setWidth;
          } else if (el.scrollLeft <= 0) {
            el.scrollLeft += conf.setWidth;
            exactScrollLeft += conf.setWidth;
          }
        });

        function animate() {
          if (
            el.dataset.pauseAnim !== "true" &&
            !el.classList.contains("active-drag")
          ) {
            exactScrollLeft += conf.dir * conf.speed;

            if (Math.abs(el.scrollLeft - exactScrollLeft) > 2) {
              exactScrollLeft = el.scrollLeft;
            } else {
              el.scrollLeft = exactScrollLeft;
            }
          } else {
            exactScrollLeft = el.scrollLeft;
          }
          requestAnimationFrame(animate);
        }
        requestAnimationFrame(animate);
      });

      // Spatial carousel loop logic (scroll only)
      {
        const spatialCarouselEl = document.getElementById("spatial-carousel");
        if (spatialCarouselEl) {
          const spatialSetWidth = (140 + 16) * 4;
          spatialCarouselEl.scrollLeft = spatialSetWidth;
          spatialCarouselEl.addEventListener("scroll", () => {
            if (spatialCarouselEl.scrollLeft >= spatialSetWidth * 2) {
              spatialCarouselEl.scrollLeft -= spatialSetWidth;
            } else if (spatialCarouselEl.scrollLeft <= 0) {
              spatialCarouselEl.scrollLeft += spatialSetWidth;
            }
          });
        }
      }
      // Filter search results
      const searchInput = document.getElementById("search-input");
      if (searchInput) {
        searchInput.addEventListener("input", (e) => {
          const query = e.target.value.toLowerCase();
          const items = document.querySelectorAll(".search-item");
          items.forEach((item) => {
            const title = item.getAttribute("data-title") || "";
            if (title.includes(query)) {
              item.style.display = "flex";
            } else {
              item.style.display = "none";
            }
          });
        });
      }
